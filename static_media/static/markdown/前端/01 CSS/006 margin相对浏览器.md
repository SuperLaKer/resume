    <title>margin</title>
    <style type="text/css">
        
        .box{
            width: 200px;
            height: 200px;
            background-color: gold;
            /*margin: 50px auto 100px auto;!*auto左右居中*!*/
            margin: 100px auto 100px;
        }
    </style>
</head>
<body>
<div class="box"></div>
